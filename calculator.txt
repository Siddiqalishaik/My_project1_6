package proj1;
import java.util.Scanner;

public class calculator {
	public static void main(String[]args) {
		char operator;//declaring variables
		double number1,number2;
		Scanner input = new Scanner(System.in);
		
		System.out.println("Enter number : ");//input 1
		number1=input.nextDouble();
		
		System.out.println("Enter operator (+,-,*,/)");//select operator
		operator=input.next().charAt(0);
		
		System.out.println("Enter number: ");//input 2
		number2=input.nextDouble();
		 input.close();
		 
		 switch(operator) {
		 case '+':
			 System.out.println(number1+number2);
			 break;
		 case '-':
			 System.out.println(number1-number2);
			 break;
		 case '*':
			 System.out.println(number1*number2);
			 break;
		 case '/':
			 if(number2!=0) {
			 System.out.println(number1/number2);
			 break;
			 }else {
				 System.out.println("error");
			 }break;
			 
			 default:
				 
				 System.out.println("invalid operator used!!!");
			 }

		 
		 }
		
		
	}
